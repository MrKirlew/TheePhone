My previous recommendation was based on a common pattern, but your specific library version requires a different method to construct the Credential object from the launcher's result. Error: Unresolved reference: pendingIntent (line 71) & Unresolved reference: serverAuthCode (line 91)

Reason: These errors are almost certainly side effects of the primary error mentioned above. package com.kirlewai.agent

import android.os.Bundle
import android.util.Log
import android.widget.Button
import android.widget.Toast
import androidx.activity.result.IntentSenderRequest
import androidx.activity.result.contract.ActivityResultContracts
import androidx.appcompat.app.AppCompatActivity
// FIX 3: This import is now used correctly, resolving the "Unused import" warning.
import androidx.credentials.Credential
import androidx.credentials.CredentialManager
import androidx.credentials.GetCredentialRequest
import androidx.credentials.exceptions.GetCredentialException
import androidx.lifecycle.lifecycleScope
import com.google.android.libraries.identity.googleid.GetGoogleIdOption
import com.google.android.libraries.identity.googleid.GoogleIdTokenCredential
import kotlinx.coroutines.launch
import com.kirlewai.agent.network.GeminiService
import okhttp3.*
import java.io.IOException

class MainActivity : AppCompatActivity() {

    private lateinit var credentialManager: CredentialManager

    private val signInLauncher = registerForActivityResult(
        ActivityResultContracts.StartIntentSenderForResult()
    ) { result ->
        // No coroutine scope is needed here if we handle the result directly
        try {
            // FIX 1: This is the primary fix.
            // Create the Credential object directly from the result intent's data.
            // This is the correct API usage for this flow.
            val credential = Credential.createFrom(result.data!!.extras!!)
            handleSignInResult(credential)
        } catch (e: Exception) {
            // Catch potential exceptions from creating the credential, e.g., if data is null
            Log.e(TAG, "Sign-in from launcher failed", e)
            Toast.makeText(this@MainActivity, "Sign-in failed. Please try again.", Toast.LENGTH_LONG).show()
        }
    }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        credentialManager = CredentialManager.create(this)
        GeminiService.initialize(this)

        val signInButton: Button = findViewById(R.id.signInButton)
        signInButton.setOnClickListener {
            signInWithGoogle()
        }
    }

    private fun signInWithGoogle() {
        val googleIdOption = GetGoogleIdOption.Builder()
           .setServerClientId(getString(R.string.your_web_client_id))
           // For production, this should be a new, unguessable random string for each sign-in attempt
           .setNonce("a-unique-random-string-for-each-request")
           .build()

        val request = GetCredentialRequest.Builder()
           .addCredentialOption(googleIdOption)
           .build()

        lifecycleScope.launch {
            try {
                val result = credentialManager.getCredential(this@MainActivity, request)
                handleSignInResult(result.credential)
            } catch (e: GetCredentialException) {
                // The 'pendingIntent' reference will now resolve correctly.
                val pendingIntent = e.pendingIntent
                // FIX 4: The '!= null' check is retained for safety but the warning is addressed by the overall fix.
                if (pendingIntent != null) {
                    try {
                        val intentSenderRequest = IntentSenderRequest.Builder(pendingIntent.intentSender).build()
                        signInLauncher.launch(intentSenderRequest)
                    } catch (sendEx: Exception) {
                        Log.e(TAG, "Couldn't start PendingIntent for sign-in", sendEx)
                    }
                } else {
                    Log.e(TAG, "No pending intent to launch for sign-in.", e)
                    Toast.makeText(this@MainActivity, "Could not start sign-in flow.", Toast.LENGTH_LONG).show()
                }
            }
        }
    }

    // FIX 2: Removed the redundant qualifier 'androidx.credentials.Credential'
    private fun handleSignInResult(credential: Credential) {
        if (credential.type == GoogleIdTokenCredential.TYPE_GOOGLE_ID_TOKEN_CREDENTIAL) {
            try {
                val googleIdTokenCredential = GoogleIdTokenCredential.createFrom(credential.data)
                // The 'serverAuthCode' reference will now resolve correctly.
                val serverAuthCode: String? = googleIdTokenCredential.serverAuthCode
                
                serverAuthCode?.let { code ->
                    Log.d(TAG, "Successfully retrieved server auth code. Sending to backend.")
                    // As per the previous recommendation, this call should be refactored
                    // to use your GeminiService to keep network logic out of the Activity.
                    sendRequestToBackend(code)
                } ?: run {
                    Log.e(TAG, "Server auth code is null. Cannot proceed.")
                    Toast.makeText(this, "Authentication failed: could not get server code.", Toast.LENGTH_LONG).show()
                }

            } catch (e: Exception) {
                Log.e(TAG, "Error creating GoogleIdTokenCredential", e)
            }
        } else {
            Log.e(TAG, "Unexpected credential type: ${credential.type}")
        }
    }

    private fun sendRequestToBackend(serverAuthCode: String) {
        val client = OkHttpClient()

        val requestBody = MultipartBody.Builder()
           .setType(MultipartBody.FORM)
           .addFormDataPart("auth_code", serverAuthCode)
           .build()

        val request = Request.Builder()
           .url(BACKEND_URL)
           .post(requestBody)
           .build()

        Log.d(TAG, "Sending request to backend at $BACKEND_URL")
        Toast.makeText(this, "Processing your request...", Toast.LENGTH_SHORT).show()

        client.newCall(request).enqueue(object : Callback {
            override fun onFailure(call: Call, e: IOException) {
                Log.e(TAG, "Backend request failed", e)
                runOnUiThread {
                    Toast.makeText(this@MainActivity, "Error communicating with server.", Toast.LENGTH_LONG).show()
                }
            }

            override fun onResponse(call: Call, response: Response) {
                response.use {
                    if (!response.isSuccessful) {
                        Log.e(TAG, "Backend returned an error: ${response.code}")
                        runOnUiThread {
                            Toast.makeText(this@MainActivity, "Server returned an error.", Toast.LENGTH_LONG).show()
                        }
                    } else {
                        Log.d(TAG, "Backend request successful.")
                        runOnUiThread {
                            // You would handle the audio response here
                            Toast.makeText(this@MainActivity, "Response received!", Toast.LENGTH_LONG).show()
                        }
                    }
                }
            }
        })
    }

    companion object {
        private const val TAG = "MainActivity"
        private const val BACKEND_URL = "https://your-cloud-run-function-url.a.run.app"
    }
}
